CREATE TABLE Student (
G_number  : VARCHAR(50) NOT NULL,
SSN : INTEGER NOT NULL,
Transcripts : CHAR(20),
Name : CHAR(20) NOT NULL,
MAJOR : CHAR,
PRIMARY KEY(G_number)

)

CREATE TABLE PEOPLE (
SSN : INTEGER NOT NULL,
PRIMARY KEY(SSN)

)

CREATE TABLE PROFESSOR (
SSN : INTEGER NOT NULL,
Prof_NAME : CHAR(25) NOT NULL,
EMAIL : VARCHAR(25) NOT NULL
NameOfDepart : CHAR(50) REFERENCES DEPARTMENT(NameOfDepart) NOT NULL
PRIMARY KEY(SSN)
)

CREATE TABLE TEACHES(
SSN : INTEGER REFERENCES PEOPLE(SSN),
Instructor_assignment : CHAR(25),
Section_No. : INTEGER,
COURSE_ID : VARCHAR,
SALARY : INTEGER,
EVALUATION : CHAR
)

CREATE TABLE COURSE(
COURSE_ID : VARCHAR(25),
TITLE :CHAR,
CREDIT : INTEGER
NameOfDepart : CHAR(50) REFERENCES DEPARTMENT(NameOfDepart),
PREREQ_ID : INTEGER

)



CREATE TABLE SECTION(
COURSE_ID : VARCHAR(25),
TITLE :CHAR,
Course status : CHAR,
Section_No. : INTEGER,
Semester : INTEGER,
Wait_List : INTEGER,
Year : INTEGER,
Prof_NAME : CHAR,
GTA : CHAR,


)


CREATE TABLE REGISTRATION
(
G_NUMBER : VARCHAR,
COURSE_ID : INTEGER,
REG_STATUS : CHAR,
SECTION_NO.: INTEGER,
COURSE_STATUS: CHAR,

WL_RANK : CHAR,
)

CREATE TABLE COURSE_TAKEN (

G_NUMBER : VARCHAR,
COURSE_ID : INTEGER,
REG_STATUS : CHAR,
SECTION_NO.: INTEGER,
COURSE_STATUS: CHAR,
GRADE : INTEGER,

)

CREATE TABLE DEPARTMENT (
NameOfDepart : CHAR(50),
COURSE_NO. : INTEGER,


)

CREATE TABLE MAJOR (
G_NUMBER : VARCHAR,
COURSE_NO. : INTEGER
)
